<script>
async function reloadNow(){
  try{
    showToast('Reloadingâ€¦');
    await fetch('/api/reload', { method: 'POST' });
    // Process will exit; supervisor relaunches it.
    // As a UX nicety, if running without supervisor, soft-reload the page after a short delay

  }
}
