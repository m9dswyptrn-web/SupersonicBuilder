@app.route("/api/status/sound", methods=["POST"])
def api_status_sound():
    ok = request.args.get("ok", "1") in ("1","true","True","yes","y")
    msg = request.args.get("msg", "")
    vol = get_req_volume()
    res = play_ping(ok, msg, volume=vol)
    return jsonify(res)
